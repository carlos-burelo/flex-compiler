numero := 79;

print "Numero: ";
print numero;
print "\n";

if numero > 10 and numero < 60 then print "F";
else if numero >= 60 and numero < 70 then print "B\n";
else if numero >= 70 and numero < 75 then print "B+\n";
else if numero >= 75 and numero < 80 then print "A-\n";
else if numero >= 80 and numero <= 100 then print "A+\n";
else print "Numero invalido\n";
