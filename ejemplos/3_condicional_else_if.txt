numero := 79;

imprimir "Numero: ";
imprimir numero;
imprimir "\n";

si numero > 10 y numero < 60 entonces imprimir "F";
else si numero >= 60 y numero < 70 entonces imprimir "B\n";
else si numero >= 70 y numero < 75 entonces imprimir "B+\n";
else si numero >= 75 y numero < 80 entonces imprimir "A-\n";
else si numero >= 80 y numero <= 100 entonces imprimir "A+\n";
else imprimir "Numero invalido\n";
